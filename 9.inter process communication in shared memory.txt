#include<stdio.h>
#include<sys/ipc.h>
#include<sys/shm.h>
#include<string.h>

int main()
{
    int shmid;
    char *p;

    shmid = shmget(1234,1024,0666|IPC_CREAT);
    p = shmat(shmid,NULL,0);

    strcpy(p,"Hello Shared Memory");
    printf("Data Written : %s\n",p);

    shmdt(p);

    return 0;
}
