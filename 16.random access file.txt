#include<stdio.h>

struct emp
{
    int id;
    char name[20];
};

int main()
{
    FILE *fp;
    struct emp e;

    fp=fopen("emp.dat","wb+");

    e.id=101;
    sprintf(e.name,"Ravi");

    fwrite(&e,sizeof(e),1,fp);

    rewind(fp);

    fread(&e,sizeof(e),1,fp);

    printf("ID=%d Name=%s",e.id,e.name);

    fclose(fp);

    return 0;
}
