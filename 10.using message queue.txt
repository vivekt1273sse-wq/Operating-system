#include<stdio.h>
#include<sys/ipc.h>
#include<sys/msg.h>

struct msg
{
    long type;
    char text[50];
};

int main()
{
    struct msg m;
    int id = msgget(1234,0666|IPC_CREAT);

    m.type=1;
    sprintf(m.text,"Hello Message Queue");

    msgsnd(id,&m,sizeof(m),0);
    msgrcv(id,&m,sizeof(m),1,0);

    printf("%s\n",m.text);

    return 0;
}
